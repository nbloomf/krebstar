Buffers

A buffer is just a zipped finger tree with value monoid MeasureText w t for some width paramter w and tab stop parameter t.

newtype Buffer w t a = Buffer
  { unBuffer :: FingerTreeZip (MeasureText w t) a
  } deriving Eq

instance
  ( IsWidth w, IsTab t, Valued (MeasureText w t) a
  ) => Valued (MeasureText w t) (Buffer w t a)
  where
    value = value . integrateFTZ . unBuffer

Immediately we can give Buffer a Tape instance:

liftBuffer
  :: (    FingerTreeZip (MeasureText w t) a
       -> FingerTreeZip (MeasureText w t) a
     )
  -> Buffer w t a -> Buffer w t a
liftBuffer f = Buffer . f . unBuffer

instance
  ( IsWidth w, IsTab t, Valued (MeasureText w t) a

Constructors and destructors

Our buffer constructors come in two flavors: constructors prefixed with def take the type parameters as inputs, while those prefixed with mk do not.

mkEmptyBuffer
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a
mkEmptyBuffer = Buffer
  { unBuffer = emptyFTZ
  }

mkBuffer
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => [a] -> Buffer w t a
mkBuffer = mkTape

mkBufferFocus
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => [a] -> a -> [a] -> Buffer w t a
mkBufferFocus = mkTapeFocus

The def variants will mostly be used for testing or in the shell.

defEmptyBuffer
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Proxy w -> Proxy t -> Buffer w t a
defEmptyBuffer _ _ = mkEmptyBuffer

defBuffer
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Proxy w -> Proxy t -> [a] -> Buffer w t a
defBuffer _ _ = mkBuffer

defBufferFocus
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Proxy w -> Proxy t -> [a] -> a -> [a] -> Buffer w t a
defBufferFocus _ _ = mkBufferFocus
mkBufferFocus'
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> a -> Buffer w t a -> Buffer w t a
mkBufferFocus' (Buffer as) x (Buffer bs) = Buffer $
  FingerTreeZip $ Just (integrateFTZ as, x, integrateFTZ bs)
charBuffer
  :: ( IsWidth w, IsTab t
     , Valued (MeasureText w t) a, IsChar a )
  => Proxy w -> Proxy t -> [Char]
  -> Buffer w t a
charBuffer _ _ as = mkBuffer (map fromChar as)
charBufferFocus
  :: ( IsWidth w, IsTab t
     , Valued (MeasureText w t) a, IsChar a )
  => Proxy w -> Proxy t -> [Char] -> Char -> [Char]
  -> Buffer w t a
charBufferFocus _ _ as x bs = mkBufferFocus
  (map fromChar as) (fromChar x) (map fromChar bs)
unBufferFocus
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> Maybe ([a], a, [a])
unBufferFocus buf =
  case unFingerTreeZip $ unBuffer buf of
    Nothing -> Nothing
    Just (as, x, bs) ->
      Just (toList as, x, toList bs)

And we can give a handy Show instance:

instance
  ( Show a, IsWidth w, IsTab t
  ) => Show (Buffer w t a)
  where
    show w =
      case unFingerTreeZip $ unBuffer w of
        Nothing -> "mkEmptyBuffer"
        Just (as, x, bs) -> concat
          [ "defBufferFocus "
          , showWidth (Proxy :: Proxy w), " "
          , showTab (Proxy :: Proxy t), " "
          , show $ toList as, " ("
          , show x, ") ", show $ toList bs
          ]

Queries

Functions for reifying the type level parameters:

bufferWidth
  :: forall w t a
   . ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> Int
bufferWidth _ = toWidth (Proxy :: Proxy w)

bufferTabStop
  :: forall w t a
   . ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> Int
bufferTabStop _ = toTab (Proxy :: Proxy t)

And functions for extracting parts of the value of a buffer which don’t involve the type parameters.

bufferLineCol
  :: forall w t a
   . ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> LineCol
bufferLineCol (Buffer buf) =
  let m = value buf :: MeasureText w t in
  logicalOffset m

bufferBytes
  :: forall w t a
   . ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> Int
bufferBytes (Buffer buf) =
  let m = value buf :: MeasureText w t in
  byteCount m
bufferHeadPosition
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> MeasureText w t
bufferHeadPosition buf =
  case unFingerTreeZip $ unBuffer buf of
    Nothing -> mempty
    Just (as, _, _) -> value as
bufferLastPosition
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> MeasureText w t
bufferLastPosition = value
bufferHeadLineCol
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> LineCol
bufferHeadLineCol =
  logicalOffset . bufferHeadPosition
bufferHeadScreenCoords
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> (Int, Int)
bufferHeadScreenCoords =
  screenCoords . bufferHeadPosition
bufferHeadScreenOffset
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> ScreenOffset w t
bufferHeadScreenOffset =
  screenOffset . bufferHeadPosition
bufferLastScreenCoords
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> (Int, Int)

  screenCoords . bufferLastPosition

It will also be handy to be able to resize the buffer in two different ways, depending on whether we want to specify the new parameters statically or dynamically.

resizeBuffer
  :: ( IsWidth w1, IsTab t1, Valued (MeasureText w1 t1) a
     , IsWidth w2, IsTab t2, Valued (MeasureText w2 t2) a )
  => Buffer w1 t1 a -> Buffer w2 t2 a
resizeBuffer (Buffer x) = Buffer (remeasureFTZ x)

adjustBuffer
  :: ( IsWidth w1, IsTab t1, Valued (MeasureText w1 t1) a
     , IsWidth w2, IsTab t2, Valued (MeasureText w2 t2) a )
  => Proxy w2 -> Proxy t2
  -> Buffer w1 t1 a -> Buffer w2 t2 a
adjustBuffer _ _ = resizeBuffer

Sized Buffers

The Buffer type is parameterized by two type level natural numbers. This is necessary to make the monoid instance for MeasureText work out nicely (and statically), but will be cumbersome once we’re juggling several text buffers at once. Consumers of this code need to know that a buffer has a well defined width and tab stop, but don’t really want to care about the types used to express this.

What we want is to be able to hide the w and t type parameters. We can do this with the following existential type, SizedBuffer.

data SizedBuffer a = forall w t.
  ( IsWidth w, IsTab t, Valued (MeasureText w t) a
  ) => SizedBuffer (Buffer w t a)

Note that the type variables w and t are bound on the right hand side of the definition, so they cannot “leak out”. Code consuming SizedBuffers cannot know about or do anything with w and t beyond what is allowed by the IsWidth and IsTab constraints. This is really powerful! Buffer gets the benefits of very expressive types, and SizedBuffer acts like a guard giving the outside world the ability to ignore those types.

instance
  ( Eq a
  , forall w t. (IsWidth w, IsTab t)
      => Valued (MeasureText w t) a
  ) => Eq (SizedBuffer a)
  where
    (SizedBuffer b1) == (SizedBuffer b2) = and
      [ (bufferWidth b1) == (bufferWidth b2)
      , (bufferTabStop b1) == (bufferTabStop b2)
      , (adjustBuffer nat8 nat2 b1)
          == (adjustBuffer nat8 nat2 b2)
      ]

instance
  ( Show a
  ) => Show (SizedBuffer a)
  where
    show (SizedBuffer buf) = 
      "SizedBuffer (" ++ show buf ++ ")"
emptySizedBuffer
  :: Int -> Int -> SizedBuffer Glyph
emptySizedBuffer width tab =
  withWidth width $ \(_ :: Proxy w) ->
    withTab tab $ \(_ :: Proxy t) ->
      SizedBuffer (mkEmptyBuffer :: Buffer w t Glyph)
shapeSizedBuffer
  :: Int -> Int
  -> SizedBuffer Glyph
  -> SizedBuffer Glyph
shapeSizedBuffer width tab (SizedBuffer buf) =
  withWidth width $ \(_ :: Proxy w) ->
    withTab tab $ \(_ :: Proxy t) ->
      SizedBuffer (resizeBuffer buf :: Buffer w t Glyph)
alterSizedBuffer
  :: (forall w t. (IsWidth w, IsTab t, Valued (MeasureText w t) a)
      => Buffer w t a -> Buffer w t a)
  -> SizedBuffer a
  -> SizedBuffer a
alterSizedBuffer f (SizedBuffer buf) = SizedBuffer (f buf)
querySizedBuffer
  :: (forall w t. (IsWidth w, IsTab t, Valued (MeasureText w t) a)
      => Buffer w t a -> b)
  -> SizedBuffer a
  -> b
querySizedBuffer f (SizedBuffer buf) = f buf

applyBufferHeadScreenOffset xy buf =

bufferFocusValue
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Buffer w t a -> MeasureText w t
bufferFocusValue buf =
  case unFingerTreeZip $ unBuffer buf of

– | Generic buffer splitting based on @split@ for – @FingerTree@s.

splitBuffer
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => (MeasureText w t -> Bool)
  -> Buffer w t a -> Maybe (Buffer w t a)
splitBuffer p buf =
  let xs = integrateFTZ $ unBuffer buf in
  case splitFT p xs of
    Nothing -> Nothing
    Just zs -> Just $ Buffer $ FingerTreeZip $ Just zs

– | Move the read head to or immediately after – the given line and column offset.

splitBufferAtLineCol
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => LineCol -> Buffer w t a -> Buffer w t a
splitBufferAtLineCol lc buf =
  case splitBuffer (afterLineCol lc) buf of
    Nothing -> lastMove buf
    Just xs -> xs
afterLineCol
  :: LineCol -> MeasureText w t -> Bool
afterLineCol lc m =
  lc < logicalOffset m
splitBufferAtScreenCoords
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => (Int, Int) -> Buffer w t a -> Buffer w t a
splitBufferAtScreenCoords x buf =
  case splitBuffer (afterScreenCoords x) buf of
    Nothing -> lastMove buf
    Just xs -> xs
afterScreenCoords
  :: forall w t
   . ( IsWidth w, IsTab t )
  => (Int, Int) -> MeasureText w t -> Bool
afterScreenCoords (u,v) m =
  let
    (h,k) = applyScreenOffset (screenOffset m) (0,0)
  in (v < k) || ((v == k) && (u < h))
seekScreenCoords
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => (Int, Int) -> Buffer w t a -> (Int, Int)
seekScreenCoords z =
  bufferHeadScreenCoords . splitBufferAtScreenCoords z
splitBufferAtScreenLine
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Int -> Buffer w t a -> Buffer w t a
splitBufferAtScreenLine k buf =
  case splitBuffer (afterScreenLine k) buf of
    Nothing -> lastMove buf
    Just xs -> if k == 0 then xs else headMoveR xs
afterScreenLine
  :: forall w t
   . ( IsWidth w, IsTab t )
  => Int -> MeasureText w t -> Bool
afterScreenLine k m =
  let
    offset = screenOffset m
    (u,v) = applyScreenOffset offset (0,0)
  in (k < v) || ((k == v) && (0 <= u))

Rendering

viewLines
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Int -- top screen line
  -> Int -- view height
  -> Buffer w t a
  -> ( FingerTree (MeasureText w t) a
     , [(FingerTree (MeasureText w t) a, MeasureText w t)]
     , FingerTree (MeasureText w t) a )
viewLines t h buf =
  let
    Buffer (FingerTreeZip w) =
      splitBufferAtScreenLine t buf
  in case w of
    Nothing ->
      (mempty, [], mempty)
    Just (as,x,bs) ->
      let
        (us, vs) =
          takeLines 0 [] (cons x bs) (value as)
      in (as, us, vs)
  where
    takeLines k us vs m =
      if (isEmptyFT vs) || (k >= h)
        then (reverse us, vs)
        else case splitTree (afterScreenLine (t+k+1)) m vs of
          Nothing -> (reverse us, vs)
          Just (as,x,bs) ->
            takeLines (k+1)
              ((snoc x as, m):us)
              bs (m <> value (snoc x as))
splitScreenLines
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => Int -- top screen line
  -> Int -- view height
  -> Buffer w t a
  -> [(FingerTree (MeasureText w t) a, MeasureText w t)]
splitScreenLines t h buf =
  let (_,xs,_) = viewLines t h buf
  in xs
renderBuffer
  :: ( IsWidth w, IsTab t, Valued (MeasureText w t) a )
  => BufferRenderSettings
  -> Int -- top screen line
  -> Int -- view height
  -> Buffer w t a
  -> ([Maybe Int], [[a]])
renderBuffer _ t h buf =
  unzip $
  map
    (\(as, m) ->
      let LineCol l c = logicalOffset m in
      (if c == 0 then Just l else Nothing, as)) $
  map
    (\(as, m) -> (toList as, m)) $
  splitScreenLines t h buf
