instance Arbitrary TextBoxAction where
  arbitrary = oneof
    [ TextBoxInsertAfter <$> arbitrary
    , TextBoxInsertManyAfter <$> arbitrary
    , return TextBoxCursorDown
    , return TextBoxCursorUp
    , return TextBoxCursorRight
    , return TextBoxCursorLeft
    ]
prop_cursor_stays_in_bounds
  :: (Int, Int) -> Int -> [TextBoxAction] -> Property
prop_cursor_stays_in_bounds (x,y) tab acts =
  (x >= 3) && (y >= 1) && (x > tab) ==>
  cursorInBounds $ alterTextBox acts $ initTextBox (x,y) tab

– Assertion Helpers

cursorIs
  :: TextBox -> (Int, Int) -> Property
cursorIs box xy =
  xy === (textboxCursor box)
heightIs
  :: TextBox -> Int -> Property
heightIs box h =
  h === (textboxHeight box)
cursorInBounds
  :: TextBox -> Bool
cursorInBounds box =
  let
    (x,y) = textboxCursor box
    l = textboxOffset box
    h = textboxHeight box
    w = textboxWidth box
  in if x < 0
    then error $ concat [ "unexpected: (x = ", show x , ") < 0" ]
    else if w <= x
      then error $ concat [ "unexpected: (x = ", show x , ") >= (w = ", show w, ")" ]
        else if y < l
          then error $ concat [ "unexpected: (y = ", show y , ") < (l = ", show l, ")" ]
            else if h+l <= y
              then error $ concat [ "unexpected: (y = ", show y , ") >= (h+l = ", show h, " + ", show l, ")" ]
                else True

– Insert one character

prep_insert_one_character
  :: Ascii
  -> DebugTextBox
prep_insert_one_character (Ascii c) =
  debugTextBox (15,10) 4 $
    [ TextBoxInsertAfter c
    ]
prop_textbox_insert_one_char_labels
  :: Ascii -> Property
prop_textbox_insert_one_char_labels c =
  let
    DebugTextBox labels lines box =
      prep_insert_one_character c
  in labels === [Just 0]
prop_textbox_insert_one_char_lines
  :: Ascii -> Property
prop_textbox_insert_one_char_lines x@(Ascii c) =
  let
    DebugTextBox labels lines box =
      prep_insert_one_character x
  in lines === [[c]]
prop_textbox_insert_one_char_height
  :: Ascii -> Property
prop_textbox_insert_one_char_height c =
  let
    DebugTextBox labels lines box =
      prep_insert_one_character c
  in heightIs box 10
prop_textbox_insert_one_char_cursor
  :: Ascii -> Property
prop_textbox_insert_one_char_cursor c =
  let
    DebugTextBox labels lines box =
      prep_insert_one_character c
  in cursorIs box $ case (toChar c) of
       '\n' -> (0,1)
       _    -> (1,0)

– Insert one non-wrapping line (no trailing newline)

prep_insert_one_line_no_wrap
  :: ( IsGlyph a )
  => (Int, Int) -> [NonNewline a]
  -> DebugTextBox
prep_insert_one_line_no_wrap dim cs =
  debugTextBox dim 4 $
    [ TextBoxInsertManyAfter (map (toGlyph . unNonNewline) cs)
    ]
genOneLineNoWrap
  :: ( Arbitrary a ) 
  => (Int, Int) -> Gen ((Int, Int), [a])
genOneLineNoWrap (xmax, ymax) = do
  w <- choose (2, xmax)
  h <- choose (1, ymax)
  k <- choose (1, w-1)
  cs <- vectorOf k arbitrary
  return ((w,h), cs)
shrinkOneLineNoWrap
  :: ( Arbitrary a )
  => ((Int, Int), [a]) -> [((Int, Int), [a])]
shrinkOneLineNoWrap (dim, as) =
  [ (dim, bs) | bs <- filter (not . null) $ shrink as ]
prop_textbox_insert_several_nonnewlines_no_wrap_labels
  :: forall a
   . ( IsChar a, IsGlyph a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_several_nonnewlines_no_wrap_labels _ =
  forAllShrink (genOneLineNoWrap (30,20)) shrinkOneLineNoWrap $
    \(dim, cs :: [NonNewline a]) ->
      let
        DebugTextBox labels lines box =
          prep_insert_one_line_no_wrap dim cs
      in labels === [Just 0]
prop_textbox_insert_several_nonnewlines_no_wrap_lines
  :: forall a
   . ( IsChar a, IsGlyph a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_several_nonnewlines_no_wrap_lines _ =
  forAllShrink (genOneLineNoWrap (30,20)) shrinkOneLineNoWrap $
    \(dim, cs :: [NonNewline a]) ->
      let
        DebugTextBox labels lines box =
          prep_insert_one_line_no_wrap dim cs
      in lines === [map (toGlyph . unNonNewline) cs]
prop_textbox_insert_several_nonnewlines_no_wrap_dim
  :: forall a
   . ( IsChar a, IsGlyph a, Eq a, Show a, Arbitrary a )

prop_textbox_insert_several_nonnewlines_no_wrap_dim _ =

    \(dim@(_,h), cs :: [NonNewline a]) ->
      let
        DebugTextBox labels lines box =
          prep_insert_one_line_no_wrap dim cs

prop_textbox_insert_several_nonnewlines_no_wrap_cursor

   . ( IsChar a, IsGlyph a, Eq a, Show a, Arbitrary a )

takeBy :: Int -> [a] -> [[a]]
takeBy k xs =
  let (as,bs) = splitAt k xs in
  if null bs
    then [as]
    else as : takeBy k bs
insertNewlines
  :: ( IsChar a )
  => [[a]] -> [[a]]
insertNewlines xss = case xss of
  [] -> []
  [xs] -> [xs]
  xs:yss -> (xs ++ [fromChar '\n']) : insertNewlines yss

– Insert multiple non-wrapping lines (no trailing newline and no vertical scroll)

prep_insert_some_lines_no_wrap_no_scroll
  :: ( Valued MeasureText a, IsChar a )
  => (Int, Int) -> [[NonNewline a]]
  -> ([Maybe Int], [[a]], TextBox a)
prep_insert_some_lines_no_wrap_no_scroll dim css =
  prepareTextBox dim $
    [ TextBoxInsertManyAfter
        (intercalate [fromChar '\n'] $ map (map unNonNewline) css)
    ]
genSomeLinesNoWrapNoScroll
  :: ( Arbitrary a ) 
  => (Int, Int) -> Gen ((Int, Int), [[a]])
genSomeLinesNoWrapNoScroll (xmax, ymax) = do
  w <- choose (2, xmax)
  h <- choose (2, ymax)
  u <- choose (1, w-1)
  v <- choose (1, h-1)
  css <- vectorOf v $ vectorOf u arbitrary
  return ((w,h), css)
shrinkSomeLinesNoWrapNoScroll
  :: ( Arbitrary a )
  => ((Int, Int), [[a]]) -> [((Int, Int), [[a]])]
shrinkSomeLinesNoWrapNoScroll (dim, xss) =
  case xss of
    _:_:_ -> [ (dim, zs) | zs <- filter (all (not . null)) $ filter (not . null) $ shrink xss ]
    _ -> []
prop_textbox_insert_some_lines_no_wrap_no_scroll_labels
  :: forall a
   . ( Valued MeasureText a, IsChar a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_some_lines_no_wrap_no_scroll_labels _ =
  forAllShrink (genSomeLinesNoWrapNoScroll (30,20)) shrinkSomeLinesNoWrapNoScroll $
    \(dim, css :: [[NonNewline a]]) ->
      let
        (labels, lines, box) =
          prep_insert_some_lines_no_wrap_no_scroll dim css
      in labels === [ Just i | (_,i) <- zip css [0..] ]
prop_textbox_insert_some_lines_no_wrap_no_scroll_lines
  :: forall a
   . ( Valued MeasureText a, IsChar a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_some_lines_no_wrap_no_scroll_lines _ =
  forAllShrink (genSomeLinesNoWrapNoScroll (30,20)) shrinkSomeLinesNoWrapNoScroll $
    \(dim, css :: [[NonNewline a]]) ->
      let
        (labels, lines, box) =
          prep_insert_some_lines_no_wrap_no_scroll dim css
      in lines ===
        (map (map unNonNewline) $ insertNewlines css)
prop_textbox_insert_some_lines_no_wrap_no_scroll_dim
  :: forall a
   . ( Valued MeasureText a, IsChar a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_some_lines_no_wrap_no_scroll_dim _ =
  forAllShrink (genSomeLinesNoWrapNoScroll (30,20)) shrinkSomeLinesNoWrapNoScroll $
    \(dim, css :: [[NonNewline a]]) ->
      let
        (labels, lines, box) =
          prep_insert_some_lines_no_wrap_no_scroll dim css
      in dimIs box dim
prop_textbox_insert_some_lines_no_wrap_no_scroll_cursor
  :: forall a
   . ( Valued MeasureText a, IsChar a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_some_lines_no_wrap_no_scroll_cursor _ =
  forAllShrink (genSomeLinesNoWrapNoScroll (30,20)) shrinkSomeLinesNoWrapNoScroll $
    \(dim, css :: [[NonNewline a]]) ->
      let
        (labels, lines, box) =
          prep_insert_some_lines_no_wrap_no_scroll dim css
      in cursorIs box (length (last css), (length css) - 1)

– Insert one line that wraps once and does not scroll

prep_insert_one_lines_wrap_once_no_scroll
  :: ( Valued MeasureText a, IsChar a )
  => (Int, Int) -> [NonNewline a]
  -> ([Maybe Int], [[a]], TextBox a)
prep_insert_one_lines_wrap_once_no_scroll dim css =
  prepareTextBox dim $
    [ TextBoxInsertManyAfter (map unNonNewline css)
    ]
genOneLineOneWrap
  :: ( Arbitrary a ) 
  => (Int, Int) -> Gen ((Int, Int), [a])
genOneLineOneWrap (xmax, ymax) = do
  w <- choose (2, xmax)
  h <- choose (2, ymax)
  k <- choose (w+1, 2*w-1)
  cs <- vectorOf k arbitrary
  return ((w,h), cs)
shrinkOneLineOneWrap
  :: ( Arbitrary a )
  => ((Int, Int), [a]) -> [((Int, Int), [a])]
shrinkOneLineOneWrap (dim, as) =
  [ (dim, bs) | bs <- filter (\z -> (fst dim) < (length z)) $ shrink as ]
prop_textbox_insert_one_line_one_wrap_labels
  :: forall a
   . ( Valued MeasureText a, IsChar a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_one_line_one_wrap_labels _ =
  forAllShrink (genOneLineOneWrap (30,20)) shrinkOneLineOneWrap $
    \(dim, cs :: [NonNewline a]) ->
      let
        (labels, lines, box) =
          prep_insert_one_lines_wrap_once_no_scroll dim cs
      in labels === [Just 0, Nothing]
prop_textbox_insert_one_line_one_wrap_lines
  :: forall a
   . ( Valued MeasureText a, IsChar a, Eq a, Show a, Arbitrary a )
  => Proxy a -> Property
prop_textbox_insert_one_line_one_wrap_lines _ =

      let
        (labels, lines, box) =
          prep_insert_one_lines_wrap_once_no_scroll dim cs
